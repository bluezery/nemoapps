SET(CONFXML "rdp.conf")
SET(APPNAME "nemordp")
SET(EXP_ICON_DIR "${ICON_DIR}/rdp")
SET(EXP_IMG_DIR "${IMG_DIR}/rdp")
ADD_DEFINITIONS(-DAPPNAME="${APPNAME}")
ADD_DEFINITIONS(-DCONFXML="${CONFXML}")
ADD_DEFINITIONS(-DEXP_ICON_DIR="${EXP_ICON_DIR}")
ADD_DEFINITIONS(-DEXP_IMG_DIR="${EXP_IMG_DIR}")

PKG_CHECK_MODULES(FREERDP REQUIRED freerdp2 freerdp-client2 winpr2)
INCLUDE_DIRECTORIES(${FREERDP_INCLUDE_DIRS})

LINK_DIRECTORIES(${FREERDP_LIBRARY_DIRS})

ADD_EXECUTABLE(${APPNAME}
    rdp.c)
ADD_DEPENDENCIES(${APPNAME} nemoutil nemowidget)

TARGET_LINK_LIBRARIES(${APPNAME}
    nemoutil
    nemowidget
    ${NEMOAPPS_COMMON_LIBRARIES}
    ${FREERDP_LIBRARIES}
	)

INSTALL(TARGETS ${APPNAME}
    DESTINATION bin
    )

CONFIGURE_FILE(${CONFXML}.in ${CONFXML} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML}
    DESTINATION ${ETC_DIR}
    )
