SET(CONFXML "test_player.conf")
SET(APPNAME "nemotest_player")

ADD_DEFINITIONS(-DAPPNAME="${APPNAME}")
ADD_DEFINITIONS(-DCONFXML="${CONFXML}")

ADD_EXECUTABLE(${APPNAME}
    player.c
    $<TARGET_OBJECTS:util>
    $<TARGET_OBJECTS:widget>
    $<TARGET_OBJECTS:gsthelper>
	)
ADD_DEPENDENCIES(${APPNAME} util)

TARGET_LINK_LIBRARIES(${APPNAME}
    ${NEMOAPPS_COMMON_LIBRARIES}
    ${GSTREAMER_LIBRARIES}
	)

INSTALL(TARGETS ${APPNAME}
    DESTINATION bin
    )

CONFIGURE_FILE(${CONFXML}.in ${CONFXML} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML}
    DESTINATION ${ETC_DIR}
    )
