SET(CONFXML "sketch.conf")
SET(APPNAME "nemosketch")
ADD_DEFINITIONS(-DAPPNAME="${APPNAME}")
ADD_DEFINITIONS(-DCONFXML="${CONFXML}")

ADD_EXECUTABLE(${APPNAME}
    sketch.c
    $<TARGET_OBJECTS:util>
    $<TARGET_OBJECTS:widget>
	)
ADD_DEPENDENCIES(${APPNAME} util widget)

TARGET_LINK_LIBRARIES(${APPNAME}
    ${NEMOAPPS_COMMON_LIBRARIES}
	)

INSTALL(TARGETS ${APPNAME}
    DESTINATION bin
    )

INSTALL(FILES ${CMAKE_CURRENT_SOURCE_DIR}/${CONFXML}
    DESTINATION ${ETC_DIR}
    )
