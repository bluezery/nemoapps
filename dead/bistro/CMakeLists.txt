SET(CONFXML "bistro.conf")
SET(APPNAME "nemobistro")
SET(BISTRO_ICON_DIR "${ICON_DIR}/bistro")
SET(BISTRO_IMG_DIR "${IMG_DIR}/bistro")
ADD_DEFINITIONS(-DAPPNAME="${APPNAME}")
ADD_DEFINITIONS(-DCONFXML="${CONFXML}")
ADD_DEFINITIONS(-DBISTRO_ICON_DIR="${BISTRO_ICON_DIR}")
ADD_DEFINITIONS(-DBISTRO_IMG_DIR="${BISTRO_IMG_DIR}")

ADD_EXECUTABLE(${APPNAME}
    bistro.c
    $<TARGET_OBJECTS:sound>
    )
ADD_DEPENDENCIES(${APPNAME} xemoutil nemowidget)

TARGET_LINK_LIBRARIES(${APPNAME}
    xemoutil
    nemowidget
    ${NEMOAPPS_COMMON_LIBRARIES}
	)

INSTALL(TARGETS ${APPNAME}
    DESTINATION bin
    )

CONFIGURE_FILE(${CONFXML}.in ${CONFXML} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML}
    DESTINATION ${ETC_DIR}
    )

INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/images/
    DESTINATION  ${BISTRO_IMG_DIR}
    )
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/icons/
    DESTINATION  ${BISTRO_ICON_DIR}
    )
