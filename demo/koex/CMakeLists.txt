SET(SUBPROJECT "koex")
SET(CONFXML "${SUBPROJECT}.conf")
SET(KOEX_ICON_DIR "${ICON_DIR}/${SUBPROJECT}")
SET(KOEX_IMG_DIR "${IMG_DIR}/${SUBPROJECT}")
SET(KOEX_LINK_DIR "${DATA_DIR}/${SUBPROJECT}")
ADD_DEFINITIONS(-DSUBPROJECT="${SUBPROJECT}")
ADD_DEFINITIONS(-DCONFXML="${CONFXML}")
ADD_DEFINITIONS(-DKOEX_ICON_DIR="${KOEX_ICON_DIR}")
ADD_DEFINITIONS(-DKOEX_IMG_DIR="${KOEX_IMG_DIR}")
ADD_DEFINITIONS(-DKOEX_LINK_DIR="${KOEX_LINK_DIR}")

CONFIGURE_FILE(${CONFXML}.in ${CONFXML} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML}
    DESTINATION ${ETC_DIR}
    )

INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/icons/
    DESTINATION  ${KOEX_ICON_DIR}
    )
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/images/
    DESTINATION  ${KOEX_IMG_DIR}
    )
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/links/
    DESTINATION  ${KOEX_LINK_DIR}
    )
