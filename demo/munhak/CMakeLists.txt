SET(APP "munhak")
SET(APPID "${NEMO_DEVID}.${APP}")
SET(APPROOT ${NEMO_ROOT}/${APPID})
SET(APPDATA_ROOT "${APPROOT}/data")

SET(CONFXML0 "background.conf")
CONFIGURE_FILE(${CONFXML0}.in ${CONFXML0} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML0}
    DESTINATION ${ETC_DIR}/${APP}/
    )

SET(CONFXML1 "overlay.conf")
CONFIGURE_FILE(${CONFXML1}.in ${CONFXML1} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML1}
    DESTINATION ${ETC_DIR}/${APP}/
    )

SET(APP_OVERLAY_DIR "${APPDATA_ROOT}/overlay")
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/overlay/
    DESTINATION  ${APP_OVERLAY_DIR}
    )

SET(APP_BG_DIR "${APPDATA_ROOT}/background")
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/background/
    DESTINATION  ${APP_BG_DIR}
    )
