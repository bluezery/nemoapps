SET(SUBPROJECT "hyundai")
SET(CONFXML "${SUBPROJECT}.conf")
SET(HYUNDAI_ICON_DIR "${ICON_DIR}/${SUBPROJECT}")
SET(HYUNDAI_IMG_DIR "${IMG_DIR}/${SUBPROJECT}")
SET(HYUNDAI_LINK_DIR "${DATA_DIR}/${SUBPROJECT}")
ADD_DEFINITIONS(-DSUBPROJECT="${SUBPROJECT}")
ADD_DEFINITIONS(-DCONFXML="${CONFXML}")
ADD_DEFINITIONS(-DHYUNDAI_ICON_DIR="${HYUNDAI_ICON_DIR}")
ADD_DEFINITIONS(-DHYUNDAI_IMG_DIR="${HYUNDAI_IMG_DIR}")
ADD_DEFINITIONS(-DHYUNDAI_LINK_DIR="${HYUNDAI_LINK_DIR}")

CONFIGURE_FILE(${CONFXML}.in ${CONFXML} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML}
    DESTINATION ${ETC_DIR}
    )

INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/icons/
    DESTINATION  ${HYUNDAI_ICON_DIR}
    )
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/images/
    DESTINATION  ${HYUNDAI_IMG_DIR}
    )
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/links/
    DESTINATION  ${HYUNDAI_LINK_DIR}
    )
