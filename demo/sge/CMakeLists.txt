SET(SUBPROJECT "sge")
SET(CONFXML "${SUBPROJECT}.conf")
SET(SGE_ICON_DIR "${ICON_DIR}/${SUBPROJECT}")
SET(SGE_IMG_DIR "${IMG_DIR}/${SUBPROJECT}")
SET(SGE_LINK_DIR "${DATA_DIR}/${SUBPROJECT}")
ADD_DEFINITIONS(-DSUBPROJECT="${SUBPROJECT}")
ADD_DEFINITIONS(-DCONFXML="${CONFXML}")
ADD_DEFINITIONS(-DSGE_ICON_DIR="${SGE_ICON_DIR}")
ADD_DEFINITIONS(-DSGE_IMG_DIR="${SGE_IMG_DIR}")
ADD_DEFINITIONS(-DSGE_LINK_DIR="${SGE_LINK_DIR}")

CONFIGURE_FILE(${CONFXML}.in ${CONFXML} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML}
    DESTINATION ${ETC_DIR}
    )

INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/icons/
    DESTINATION  ${SGE_ICON_DIR}
    )
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/images/
    DESTINATION  ${SGE_IMG_DIR}
    )
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/links/
    DESTINATION  ${SGE_LINK_DIR}
    )
