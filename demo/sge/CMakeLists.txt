SET(APP "sge")
SET(APPID "${NEMO_DEVID}.${APP}")
SET(APPROOT ${NEMO_ROOT}/${APPID})
SET(APPDATA_ROOT "${APPROOT}/data")

SET(CONFXML0 "card.conf")
CONFIGURE_FILE(${CONFXML0}.in ${CONFXML0} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML0}
    DESTINATION ${ETC_DIR}/${APP}/
    )

SET(CONFXML1 "background.conf")
CONFIGURE_FILE(${CONFXML1}.in ${CONFXML1} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML1}
    DESTINATION ${ETC_DIR}/${APP}/
    )

SET(APP_ICON_DIR "${APPDATA_ROOT}/icons")
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/icons/
    DESTINATION  ${APP_ICON_DIR}
    )

SET(APP_IMG_DIR "${APPDATA_ROOT}/images")
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/images/
    DESTINATION  ${APP_IMG_DIR}
    )

SET(APP_LINK_DIR "${APPDATA_ROOT}/links")
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/links/
    DESTINATION  ${APP_LINK_DIR}
    )
