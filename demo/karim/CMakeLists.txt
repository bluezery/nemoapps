SET(CONFXML "karim.conf")
SET(APPNAME "nemokarim")
SET(KARIM_ICON_DIR "${ICON_DIR}/karim")
SET(KARIM_IMG_DIR "${IMG_DIR}/karim")
SET(KARIM_LINK_DIR "${DATA_DIR}/karim")
ADD_DEFINITIONS(-DAPPNAME="${APPNAME}")
ADD_DEFINITIONS(-DCONFXML="${CONFXML}")
ADD_DEFINITIONS(-DKARIM_ICON_DIR="${KARIM_ICON_DIR}")
ADD_DEFINITIONS(-DKARIM_IMG_DIR="${KARIM_IMG_DIR}")
ADD_DEFINITIONS(-DKARIM_LINK_DIR="${KARIM_LINK_DIR}")

ADD_EXECUTABLE(${APPNAME}
    karim.c
	)
ADD_DEPENDENCIES(${APPNAME} nemoutil nemowidget)

TARGET_LINK_LIBRARIES(${APPNAME}
    nemoutil
    nemowidget
    ${NEMOAPPS_COMMON_LIBRARIES}
	)

INSTALL(TARGETS ${APPNAME}
    DESTINATION bin
    )

CONFIGURE_FILE(${CONFXML}.in ${CONFXML} @ONLY)
INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/${CONFXML}
    DESTINATION ${ETC_DIR}
    )

INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/icons/
    DESTINATION  ${KARIM_ICON_DIR}
    )

INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/images/
    DESTINATION  ${KARIM_IMG_DIR}
    )
INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/links/
    DESTINATION  ${KARIM_LINK_DIR}
    )
